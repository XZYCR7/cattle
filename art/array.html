# 数组去重

思路一：for 循环

const unique = arr => {
  for (let i = 0; i < arr.length - 1; i++) {
    for (let j = i + 1; i < arr.length; j++) {
      if (arr[i] === arr[j]) {
        arr.splice(j, 1);
        j -= 1;
        arr.length -= 1;
      }
    }
  }
}

思路二：for ... of + includes

const unique = arr => {
  const res = []
  for (const item of arr) !res.includes(item) && res.push(item)
  return res
}

思路三：filter + indexOf

const unique = arr => {
  return arr.filter((item, index) => {
    return arr.indexOf(item) === index
  })
}

思路四：sort

const unique = arr => {
  arr.sort()
  const res = [arr[0]]
  for (let i = 1, len = arr.length; i < len; i++) {
    arr[i] !== arr[i - 1] && res.push(arr[i])
  }
  return res
}

思路五：Set

const unique = arr => [...new Set(arr)]

思路六：for ... of + Object

const unique = arr => {
  const res = []
  const obj = {}
  for (const item of arr) {
    if (!obj[item]) {
      res.push(item)
      obj[item] = true
    }
  }
  return res
}

# 数组扁平化

核心：迭代数组，如果数组项是数组，那么递归，否则与之前的结果合并。

思路一：toString + split

const flatten = arr => arr.toString().split(',').map(item => Number(item))

思路二：join + split

const flatten = arr => arr.join().split(',').map(item => parseInt(item))

思路三：递归

const flatten = arr => {
  const res = []
  arr.map(item => {
    Array.isArray(item) ? res = res.concat(flatten(item)) : res.push(item)
  })
  return res
}

思路四：reduce

const flatten = arr => {
  return arr.reduce((res, item) => {
    return res.concat(Array.isArray(item) ? flatten(item) : item)
  }, [])
}

思路五：扩展运算符

const flatten = arr => {
  while (arr.some(item => Array.isArray(item))) {
    arr = [].concat(...arr)
  }
  return arr
}
